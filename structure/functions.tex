% custom defined functions
% ~~~~~~~~~~~~~~~~~~~~~~~~

% insert a blank page
\newcommand\blankpage{
    \null{}
    \thispagestyle{empty}
    \addtocounter{page}{-1}
    \newpage
    }

% create a quote function
\newcommand\makequote[3]{
    \begin{center}
        \Huge{\textbf{\textcolor{sections}{``~~}}}\\
            \LARGE{#1}\\
        \vspace{0.3cm}
        \Huge{\textbf{\textcolor{sections}{~~''}}}
    \end{center}
    \vspace{-0.7cm} \hspace{8cm}
    \large{ --- #2\ifthenelse{\isempty{#3}}{}{, \textit{#3}}}
    }
