% insert a blank page
\newcommand\blankpage{
    \null{}
    \thispagestyle{empty}
    \addtocounter{page}{-1}
    \newpage
    }

% format a quote
\newcommand\makequote[3]{
    \begin{center}
        \Huge{\textbf{\textcolor{GOTOlightblue}{``~~}}}\\
            \LARGE{#1}\\
        \vspace{0.3cm}
        \Huge{\textbf{\textcolor{GOTOlightblue}{~~''}}}
    \end{center}
    \vspace{-0.7cm} \hspace{8cm}
    \large{ --- #2\ifthenelse{\isempty{#3}}{}{, \textit{#3}}}
    }

    \noindent{\color{GOTOlightblue}\rule{\textwidth}{1.5pt}}
    \vspace{-0.7cm}
    \minitoc{}
    \vspace{-0.7cm}
    \noindent{\color{GOTOlightblue}\rule{\textwidth}{1.5pt}}
    \bigskip
    \end{singlespacing}
    }
